Adresse v.s. valeur numrique

Les adresses et les valeurs numriques sont diffrencie en notant un # devant ces dernires :

Exemple : MOVE #17, HFF transfert la valeur numrique 17 ladresse HFF

Les adresses sont encodes sur 12 bits et enregistre en mmoire sur deux octets.

Les valeurs numriques sont encodes sur 8 bits et donc peuvent prendre des valeurs entre 0 et 255 non sign ou entre -128 et 127 pour les valeurs signes.

SCREENCOPY = H'20F ; adresse de la copie d'cran

SCREENSIZE = 16 ; taille de l'cran en octets (original = 96 ; max 128)

 

MOVE #0, A

CALL PUTMOTIF

 

CALL SCREENTOCOPY

 

MOVE #1, X

MOVE #2, Y

CALL NEIGHBOURS

 

LOOP:

MOVE #0, X

MOVE #0, Y

 

LOOPX:

CALL TESTPIXEL

COMP #1, A

JUMP,EQ ITSALIVE

CALL NEIGHBOURS

COMP #3, B

JUMP,EQ BIRTH

JUMP ENDLOOPX

 

ITSALIVE:

CALL NEIGHBOURS

COMP #2, B

JUMP,LO DIE

COMP #3, B

JUMP,LS ENDLOOPX

 

DIE:

CALL CLRPIXEL

JUMP ENDLOOPX

 

BIRTH:

CALL SETPIXEL

 

ENDLOOPX:

INC X

COMP #32, X

JUMP,LO LOOPX

CLR X

 

INC Y

COMP #24, Y

JUMP,LO LOOPX

 

CALL COPYTOSCREEN

JUMP LOOP

 

; Copie l'cran vers la copie d'cran.

SCREENTOCOPY:

MOVE #0, X

MOVE #0, Y

MOVE #SCREENSIZE, B

LOOP2:

MOVE _BITMAP+{X}+{Y}, A

MOVE A, SCREENCOPY+{X}+{Y}

INC X

DEC B

JUMP,NE LOOP2

RET

 

; Copie la copie d'cran vers l'cran.

COPYTOSCREEN:

MOVE #0, X

MOVE #0, Y

MOVE #SCREENSIZE, B

LOOP1:

MOVE SCREENCOPY+{X}+{Y}, A

MOVE A, _BITMAP+{X}+{Y}

INC X

DEC B

JUMP,NE LOOP1

RET

; Teste un pixel dans l'cran.

; in X,Y coordonnes

; out A 1 => pixel allum

; 0 => pixel teint

TESTPIXEL:

PUSH B

PUSH X

PUSH Y

 

MOVE #0, A

 

CALL WRAPXY

 

AND #H'1F, Y

RL Y

RL Y

MOVE X, B

XOR #H'07, B

RR X

RR X

RR X

AND #H'3, X

 

TEST _BITMAP+{X}+{Y} :B

JUMP,EQ TESTPIXEL1

INC A

 

TESTPIXEL1:

POP Y

POP X

POP B

RET

 

; Efface un pixel dans la copie d'cran.

; in X,Y coordonnes

; out A 1 => pixel allum

; 0 => pixel teint

CLRPIXEL:

PUSH B

PUSH X

PUSH Y

AND #H'1F, Y

RL Y

RL Y

MOVE X, B

XOR #H'07, B

RR X

RR X

RR X

AND #H'3, X

MOVE #0, A

TCLR SCREENCOPY+{X}+{Y} :B

POP Y

POP X

POP B

RET

 

; Allume un pixel dans la copie d'cran.

; in X,Y coordonnes

; out A 1 => pixel allum

; 0 => pixel teint

SETPIXEL:

PUSH B

PUSH X

PUSH Y

AND #H'1F, Y

RL Y

RL Y

MOVE X, B

XOR #H'07, B

RR X

RR X

RR X

AND #H'3, X

MOVE #0, A

TSET SCREENCOPY+{X}+{Y} :B

 

POP Y

POP X

POP B

RET

 

; Wrap des coordonnes.

; in X,Y coordonnes

; out X,Y coordonnes wrappes

WRAPXY:

COMP #31, X

JUMP,LS XOK

 

COMP #128, X

JUMP,LS XOK1

MOVE #31, X

JUMP XOK

 

XOK1:

MOVE #0, X

JUMP XOK

 

XOK:

COMP #23, Y

JUMP,LS YOK

 

COMP #128, Y

JUMP,LS YOK1

MOVE #23, Y

JUMP YOK

 

YOK1:

MOVE #0, Y

YOK:

RET
